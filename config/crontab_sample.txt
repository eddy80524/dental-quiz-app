# Type=oneshot

# WorkingDirectory=/Users/utsueito/kokushi-dx-poc/dental-DX-PoC/my_llm_app

# Environment=PYTHONPATH=/Users/utsueito/kokushi-dx-poc/dental-DX-PoC/my_llm_app

# /etc/systemd/system/ranking-batch.timer:
# [Unit]
# Description=Run Weekly Ranking Batch Process daily at 3 AM
# Requires=ranking-batch.service
# 
# [Timer]
# OnCalendar=*-*-* 03:00:00
# Persistent=true
# 
# [Install]
# WantedBy=timers.target
#
